## FIX 5 — Study Waves Audio Fix (Mountain Wind + General Audio Issues)

### The Problem:
Audio is mismatching between wave types and there are audio issues especially
with Mountain Wind. The problem is likely:
1. Previous audio nodes not being fully stopped before new ones start
2. Mountain Wind using incorrect filter/oscillator settings causing distortion or wrong sound
3. AudioContext not being closed properly when switching wave types causing overlap

### Fix — Robust Audio Cleanup Function:

Every time a new wave type is selected OR the user exits Study Waves, ALL existing
audio nodes must be completely stopped and disconnected before creating new ones:

```javascript
// Global audio state
let activeNodes = [];
let activeContext = null;

function stopAllAudio() {
  // Stop all active nodes
  activeNodes.forEach(node => {
    try {
      if (node.stop) node.stop(0);
      node.disconnect();
    } catch (e) {
      // Already stopped — ignore
    }
  });
  activeNodes = [];

  // Close and nullify the AudioContext
  if (activeContext) {
    try {
      activeContext.close();
    } catch (e) {}
    activeContext = null;
  }
}

function startWave(type) {
  // ALWAYS stop everything first
  stopAllAudio();

  // Small delay to ensure cleanup is complete before starting new audio
  setTimeout(() => {
    activeContext = new (window.AudioContext || window.webkitAudioContext)();
    const nodes = createWaveSound(type, activeContext);
    activeNodes = nodes;
  }, 100);
}
```

### Fix — Each Wave Type Sound Definition:

Replace ALL wave sound implementations with these clean, correct versions:

```javascript
function createWaveSound(type, ctx) {
  const masterGain = ctx.createGain();
  masterGain.gain.value = 0.4;
  masterGain.connect(ctx.destination);
  const nodes = [masterGain];

  switch(type) {

    case "Ocean Waves": {
      // Brown noise with slow volume modulation = ocean wave rhythm
      const bufferSize = ctx.sampleRate * 4;
      const buffer = ctx.createBuffer(1, bufferSize, ctx.sampleRate);
      const data = buffer.getChannelData(0);
      let last = 0;
      for (let i = 0; i < bufferSize; i++) {
        const white = Math.random() * 2 - 1;
        last = (last + 0.02 * white) / 1.02;
        data[i] = last * 4;
      }
      const source = ctx.createBufferSource();
      source.buffer = buffer;
      source.loop = true;

      const filter = ctx.createBiquadFilter();
      filter.type = "lowpass";
      filter.frequency.value = 500;
      filter.Q.value = 0.5;

      // LFO for wave rhythm
      const lfo = ctx.createOscillator();
      const lfoGain = ctx.createGain();
      lfo.frequency.value = 0.15; // slow wave rhythm
      lfoGain.gain.value = 0.3;
      lfo.connect(lfoGain);
      lfoGain.connect(masterGain.gain);
      lfo.type = "sine";

      source.connect(filter);
      filter.connect(masterGain);
      lfo.start(0);
      source.start(0);
      nodes.push(source, filter, lfo, lfoGain);
      break;
    }

    case "Rain Storm": {
      // White noise through highpass = rain sound
      const bufferSize = ctx.sampleRate * 2;
      const buffer = ctx.createBuffer(1, bufferSize, ctx.sampleRate);
      const data = buffer.getChannelData(0);
      for (let i = 0; i < bufferSize; i++) {
        data[i] = Math.random() * 2 - 1;
      }
      const source = ctx.createBufferSource();
      source.buffer = buffer;
      source.loop = true;

      const highpass = ctx.createBiquadFilter();
      highpass.type = "highpass";
      highpass.frequency.value = 1200;

      const lowpass = ctx.createBiquadFilter();
      lowpass.type = "lowpass";
      lowpass.frequency.value = 8000;

      source.connect(highpass);
      highpass.connect(lowpass);
      lowpass.connect(masterGain);
      source.start(0);
      nodes.push(source, highpass, lowpass);
      break;
    }

    case "Fire Crackle": {
      // Filtered noise with slow random amplitude modulation = crackling fire
      const bufferSize = ctx.sampleRate * 2;
      const buffer = ctx.createBuffer(1, bufferSize, ctx.sampleRate);
      const data = buffer.getChannelData(0);
      let lastSample = 0;
      for (let i = 0; i < bufferSize; i++) {
        lastSample = (Math.random() * 2 - 1) * 0.5 + lastSample * 0.5;
        data[i] = lastSample;
      }
      const source = ctx.createBufferSource();
      source.buffer = buffer;
      source.loop = true;

      const filter = ctx.createBiquadFilter();
      filter.type = "bandpass";
      filter.frequency.value = 800;
      filter.Q.value = 0.3;

      // Crackle modulation
      const lfo = ctx.createOscillator();
      const lfoGain = ctx.createGain();
      lfo.frequency.value = 3.5;
      lfo.type = "sawtooth";
      lfoGain.gain.value = 0.2;
      lfo.connect(lfoGain);
      lfoGain.connect(masterGain.gain);

      source.connect(filter);
      filter.connect(masterGain);
      lfo.start(0);
      source.start(0);
      nodes.push(source, filter, lfo, lfoGain);
      break;
    }

    case "Space Drift": {
      // Slow sine tones layered = deep space ambient
      const freqs = [55, 82.5, 110, 165];
      freqs.forEach((freq, i) => {
        const osc = ctx.createOscillator();
        const gain = ctx.createGain();
        osc.type = "sine";
        osc.frequency.value = freq;
        gain.gain.value = 0.08;

        // Slow fade in/out for each tone
        const lfo = ctx.createOscillator();
        const lfoGain = ctx.createGain();
        lfo.type = "sine";
        lfo.frequency.value = 0.05 + i * 0.02;
        lfoGain.gain.value = 0.06;
        lfo.connect(lfoGain);
        lfoGain.connect(gain.gain);

        osc.connect(gain);
        gain.connect(masterGain);
        osc.start(0);
        lfo.start(0);
        nodes.push(osc, gain, lfo, lfoGain);
      });
      break;
    }

    case "Lo-Fi Waves": {
      // Warm chord tones with gentle tremolo = lo-fi feel
      const chordFreqs = [130.81, 164.81, 196.00, 246.94]; // C E G B
      chordFreqs.forEach((freq, i) => {
        const osc = ctx.createOscillator();
        const gain = ctx.createGain();
        osc.type = "triangle";
        osc.frequency.value = freq;
        gain.gain.value = 0.06;

        // Gentle tremolo
        const tremolo = ctx.createOscillator();
        const tremoloGain = ctx.createGain();
        tremolo.type = "sine";
        tremolo.frequency.value = 4.5;
        tremoloGain.gain.value = 0.02;
        tremolo.connect(tremoloGain);
        tremoloGain.connect(gain.gain);

        osc.connect(gain);
        gain.connect(masterGain);
        osc.start(0);
        tremolo.start(0);
        nodes.push(osc, gain, tremolo, tremoloGain);
      });
      break;
    }

    case "Mountain Wind": {
      // THIS WAS THE BROKEN ONE — rewritten correctly
      // Filtered white noise swept slowly with LFO = wind through trees
      const bufferSize = ctx.sampleRate * 3;
      const buffer = ctx.createBuffer(2, bufferSize, ctx.sampleRate); // stereo
      for (let channel = 0; channel < 2; channel++) {
        const data = buffer.getChannelData(channel);
        for (let i = 0; i < bufferSize; i++) {
          data[i] = Math.random() * 2 - 1;
        }
      }
      const source = ctx.createBufferSource();
      source.buffer = buffer;
      source.loop = true;

      // Bandpass filter — narrow band = wind tone
      const bandpass = ctx.createBiquadFilter();
      bandpass.type = "bandpass";
      bandpass.frequency.value = 600;
      bandpass.Q.value = 1.5;

      // Second filter for shimmer
      const highshelf = ctx.createBiquadFilter();
      highshelf.type = "highshelf";
      highshelf.frequency.value = 3000;
      highshelf.gain.value = -6; // reduce harshness

      // LFO to slowly sweep the filter frequency = gusting wind
      const lfo = ctx.createOscillator();
      const lfoGain = ctx.createGain();
      lfo.type = "sine";
      lfo.frequency.value = 0.08; // very slow sweep
      lfoGain.gain.value = 300; // sweep range in Hz
      lfo.connect(lfoGain);
      lfoGain.connect(bandpass.frequency); // modulate filter frequency

      // Volume LFO = wind gusts
      const volLfo = ctx.createOscillator();
      const volLfoGain = ctx.createGain();
      volLfo.type = "sine";
      volLfo.frequency.value = 0.12;
      volLfoGain.gain.value = 0.15;
      volLfo.connect(volLfoGain);
      volLfoGain.connect(masterGain.gain);

      source.connect(bandpass);
      bandpass.connect(highshelf);
      highshelf.connect(masterGain);

      source.start(0);
      lfo.start(0);
      volLfo.start(0);

      nodes.push(source, bandpass, highshelf, lfo, lfoGain, volLfo, volLfoGain);
      break;
    }

    default:
      break;
  }

  return nodes;
}
```
