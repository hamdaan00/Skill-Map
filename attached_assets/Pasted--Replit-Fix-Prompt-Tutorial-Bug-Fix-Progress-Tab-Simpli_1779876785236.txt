## 🐛 BUG FIX 1 — Tutorial Showing Blank on Step 3+

### The Problem
Tutorial works for steps 1 and 2 but shows a blank/empty screen from step 3 onwards
when the user clicks "Next →".

### Root Causes to Check and Fix

**Cause 1 — Target element not found in DOM**
Steps 3–7 try to spotlight a specific DOM element using `getElementById` or
`querySelector`. If the element doesn't exist yet (wrong ID, not rendered, inside
a different route), the spotlight has nothing to point to and renders blank.

Fix: Add a safety check before every step renders:
```javascript
function getStepTarget(selector) {
  const el = document.querySelector(selector);
  if (!el) {
    console.warn(`Tutorial target not found: ${selector}`);
    return null;
  }
  return el;
}

// In the step renderer:
const target = getStepTarget(step.selector);
if (!target) {
  // Skip to next step or show tooltip centered on screen as fallback
  showStepCentered(step); // fallback: show tooltip in center with no spotlight
  return;
}
```

**Cause 2 — Steps array index off by one**
The steps array might have undefined entries at index 2+. Check the steps array:
```javascript
// Wrong — gap at index 2 causes blank:
const steps = [
  { id: 1, ... },
  { id: 2, ... },
  undefined,        // ← gap causes blank screen
  { id: 4, ... },
];

// Correct — continuous array, no gaps:
const steps = [
  { id: 1, ... },
  { id: 2, ... },
  { id: 3, ... },   // ← must exist
  { id: 4, ... },
  { id: 5, ... },
  { id: 6, ... },
  { id: 7, ... },
];
```

**Cause 3 — currentStep state not updating correctly**
If `setCurrentStep` is called but the component doesn't re-render with new step data:
```javascript
// Wrong:
function nextStep() {
  currentStep++; // mutating directly — React won't re-render
}

// Correct:
function nextStep() {
  setCurrentStep(prev => prev + 1); // functional update — always correct
}
```

**Cause 4 — Steps 3–7 targeting elements on different pages/routes**
If the tutorial runs on the dashboard but step 3 targets the "Add Skill" button
which only exists on a different page — the element won't be found.

Fix: For steps that target elements not currently visible, either:
- Navigate to that page before showing the step, OR
- Use a **centered fallback tooltip** (no spotlight, just the tooltip card in center screen)

```javascript
// Centered fallback for steps where target is not on current page:
const CENTERED_STEPS = [1, 7]; // steps that are always centered
const isCentered = CENTERED_STEPS.includes(currentStep) || !targetElement;

if (isCentered) {
  // Render overlay + centered tooltip card, no spotlight hole
} else {
  // Render spotlight around targetElement
}
```

**Robust step renderer:**
```javascript
function TutorialOverlay({ onComplete, onSkip }) {
  const [currentStep, setCurrentStep] = useState(0);
  const [targetRect, setTargetRect] = useState(null);

  const step = TUTORIAL_STEPS[currentStep];

  useEffect(() => {
    if (!step) return;
    if (step.centered || !step.selector) {
      setTargetRect(null);
      return;
    }
    // Small delay to ensure DOM is ready
    const timer = setTimeout(() => {
      const el = document.querySelector(step.selector);
      if (el) {
        const rect = el.getBoundingClientRect();
        setTargetRect(rect);
        el.style.position = "relative";
        el.style.zIndex = "10001";
      } else {
        // Element not found — show centered fallback
        setTargetRect(null);
      }
    }, 100);
    return () => {
      clearTimeout(timer);
      // Clean up z-index on previous element
      if (step.selector) {
        const el = document.querySelector(step.selector);
        if (el) { el.style.zIndex = ""; el.style.position = ""; }
      }
    };
  }, [currentStep]);

  function handleNext() {
    if (currentStep >= TUTORIAL_STEPS.length - 1) {
      onComplete();
    } else {
      setCurrentStep(prev => prev + 1);
    }
  }

  function handleBack() {
    if (currentStep > 0) setCurrentStep(prev => prev - 1);
  }

  if (!step) return null; // safety check — never render blank

  const isCentered = step.centered || !targetRect;

  return (
    <div style={{
      position: "fixed", inset: 0, zIndex: 10000,
      background: isCentered ? "rgba(0,0,0,0.8)" : "transparent",
    }}>
      {/* Spotlight overlay for non-centered steps */}
      {!isCentered && targetRect && (
        <div style={{
          position: "fixed", inset: 0, zIndex: 10000,
          background: "rgba(0,0,0,0.75)",
          // Cut out hole using clip-path or box-shadow
          boxShadow: `0 0 0 9999px rgba(0,0,0,0.75)`,
          top: targetRect.top - 8,
          left: targetRect.left - 8,
          width: targetRect.width + 16,
          height: targetRect.height + 16,
          borderRadius: 12,
          background: "transparent",
          outline: "9999px solid rgba(0,0,0,0.75)",
          boxShadow: "0 0 0 4px #00d4ff, 0 0 30px rgba(0,212,255,0.4)",
          animation: "tutorialPulse 2s ease-in-out infinite",
          pointerEvents: "none",
        }} />
      )}

      {/* Tooltip card */}
      <div style={{
        position: "fixed",
        zIndex: 10001,
        ...(isCentered ? {
          top: "50%", left: "50%",
          transform: "translate(-50%, -50%)",
        } : targetRect ? {
          // Position below target, or above if near bottom
          top: targetRect.bottom + 16 > window.innerHeight - 200
            ? targetRect.top - 180
            : targetRect.bottom + 16,
          left: Math.min(
            Math.max(targetRect.left, 16),
            window.innerWidth - 316
          ),
        } : {
          top: "50%", left: "50%",
          transform: "translate(-50%, -50%)",
        }),
        width: 300,
        background: "rgba(10,15,30,0.97)",
        border: "1px solid rgba(0,212,255,0.4)",
        borderRadius: 16,
        padding: 24,
        boxShadow: "0 0 40px rgba(0,212,255,0.15)",
        animation: "fadeIn 0.3s ease",
      }}>
        {/* Step counter */}
        <div style={{ fontSize: 11, color: "#00d4ff", marginBottom: 12, letterSpacing: 1 }}>
          STEP {currentStep + 1} OF {TUTORIAL_STEPS.length}
        </div>

        {/* Progress dots */}
        <div style={{ display: "flex", gap: 4, marginBottom: 16 }}>
          {TUTORIAL_STEPS.map((_, i) => (
            <div key={i} style={{
              width: i === currentStep ? 20 : 6,
              height: 6, borderRadius: 3,
              background: i === currentStep ? "#00d4ff"
                : i < currentStep ? "rgba(0,212,255,0.4)"
                : "rgba(255,255,255,0.1)",
              transition: "all 0.3s ease",
            }} />
          ))}
        </div>

        {step.emoji && (
          <div style={{ fontSize: 40, marginBottom: 12, textAlign: "center" }}>{step.emoji}</div>
        )}
        <div style={{ fontWeight: 700, fontSize: 15, color: "#e2e8f0", marginBottom: 8 }}>
          {step.title}
        </div>
        <div style={{ fontSize: 13, color: "#94a3b8", lineHeight: 1.7, marginBottom: 20 }}>
          {step.message}
        </div>

        {/* Buttons */}
        <div style={{ display: "flex", gap: 8, justifyContent: "space-between", alignItems: "center" }}>
          <button onClick={onSkip}
            style={{ background: "none", border: "none", color: "#475569", fontSize: 12, cursor: "pointer", fontFamily: "Sora, sans-serif" }}>
            Skip tutorial
          </button>
          <div style={{ display: "flex", gap: 8 }}>
            {currentStep > 0 && (
              <button onClick={handleBack}
                style={{ background: "rgba(255,255,255,0.05)", border: "1px solid rgba(255,255,255,0.1)", color: "#94a3b8", padding: "8px 16px", borderRadius: 8, cursor: "pointer", fontSize: 13, fontFamily: "Sora, sans-serif" }}>

            </button>
          </div>
        </div>
      </div>
    </div>
```
